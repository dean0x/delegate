**Recommendation**: Use heap-based priority queue (O(log n) insertion)
```typescript
import { PriorityQueue } from '@datastructures-js/priority-queue';

class PriorityTaskQueue {
  private heap = new PriorityQueue<Task>((a, b) =>
    this.compareTaskPriority(a, b)
  );

  enqueue(task: Task): Result<void> {
    this.heap.enqueue(task); // O(log n)
    return ok(undefined);
  }
}
```

}
